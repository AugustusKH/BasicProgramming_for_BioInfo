#!/bin/bash

checkpalindrome()
{
case_str=$(echo "$1")
case_str=$(echo ${case_str} | sed 's/ //g')
case_str=$(echo ${case_str} | sed 's/\.//g')

len_str=$(echo ${#case_str})

reverse_str=$(
for (( i = $len_str - 1; i >= 0; i-- ))
do
        echo ${case_str:$i:1}
done
)

reverse_str=$(
echo ${reverse_str} | sed 's/ //g'
)

reverse_str=$(
echo ${reverse_str} | sed 's/\.//g'
)

if [[ "$case_str" == "$reverse_str" ]]
then
        echo "$1" is a palindrome
else
        echo "$1" is NOT a palindrome
fi
}

checkpalindrome "$1"

#!/bin/bash

max3() 
{
        [ $# -ne 3 ] && return 5
        [ $1 -gt $2 ] && { set -- $2 $1 $3; }
        [ $2 -gt $3 ] && { set -- $1 $3 $2; }
        [ $1 -gt $2 ] && { set -- $2 $1 $3; }
        local num_st=$3
        local num_nd=$2
        local num_rd=$1
        echo "$num_st" "$num_nd" "$num_rd"      
}

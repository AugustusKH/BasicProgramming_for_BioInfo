#!/bin/bash

sort3()
{
        if [[ "$4" = [!a-zA-Z] ]]
        then
                echo "Invalid variable name"
                exit 1
        else
                declare -n max3=${4:-max3}
                (( $# < 3 )) && return 4
                (( $1 > $2 )) && set -- "$2" "$1" "$3"
                (( $2 > $3 )) && set -- "$1" "$3" "$2"
                (( $1 > $2 )) && set -- "$2" "$1" "$3"

                echo $1
                echo $2
                echo $3
        fi
}
sort3 $1 $2 $3 $4
